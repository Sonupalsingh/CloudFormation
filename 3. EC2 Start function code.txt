import boto3
import logging
import botocore  # Import botocore for exception handling

region = 'us-east-1'
instances = ['correct-instance-id1', 'correct-instance-id2']  # Replace with correct instance IDs
ec2 = boto3.client('ec2', region_name=region)
logger = logging.getLogger()
logger.setLevel(logging.INFO)

def lambda_handler(event, context):
    try:
        ec2.start_instances(InstanceIds=instances)
        logger.info(f'Started instances: {instances}')
    except botocore.exceptions.ClientError as e:
        logger.error(f'Error starting instances: {e}')
